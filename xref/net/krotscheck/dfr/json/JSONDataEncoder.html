5   package net.krotscheck.dfr.json;

7   import com.fasterxml.jackson.core.JsonFactory;
8   import com.fasterxml.jackson.core.JsonGenerator;
9   import com.fasterxml.jackson.databind.ObjectMapper;
10  import net.krotscheck.dfr.AbstractDataEncoder;
11  import org.slf4j.Logger;
12  import org.slf4j.LoggerFactory;
13  
14  import java.io.IOException;
15  import java.util.Map;
16  
17  /**
18   * JSON Data Encoder, allows us to write tuples to a data file in JSON format.
19   *
20   * @author Michael Krotscheck
21   */
22  public final class JSONDataEncoder extends AbstractDataEncoder {
23  
24      /**
25       * Logger instance.
26       */
27      private static Logger logger =
28              LoggerFactory.getLogger(JSONDataEncoder.class);
29  
30      /**
31       * The JSON generation engine.
32       */
33      private JsonGenerator generator;
34  
35      /**
36       * Return the decoding mimetype which this encoder supports.
37       *
38       * @return "application/json"
39       */
40      @Override
41      public String getMimeType() {
42          return "application/json";
43      }
44  
45      /**
46       * Write a row to the file.
47       *
48       * @param row A row of data.
49       * @throws java.io.IOException Thrown if there's a problem writing to the
50       *                             destination.
51       */
52      @Override
53      protected void writeToStream(final Map<String, Object> row)
54              throws IOException {
55          if (generator == null) {
56              ObjectMapper mapper = new ObjectMapper();
57              JsonFactory factory = new JsonFactory(mapper);
58              generator = factory.createGenerator(getOutputStream());
59              generator.writeStartArray(); // [
60          }
61  
62          // Convert the row into a map
63          generator.writeObject(row);
64      }
65  
66      /**
67       * Closes the output stream.
68       */
69      @Override
70      public void dispose() {
71          try {
72              if (generator != null) {
73                  generator.writeEndArray();
74                  generator.close();
75              }
76              this.getOutputStream().close();
77          } catch (IOException ioe) {
78              logger.error("Unable to close stream", ioe);
79              logger.trace(ioe.getMessage(), ioe);
80          } finally {
81              generator = null;
82              this.setOutputStream(null);
83          }
84      }
85  }
